## Bottom Up
```py
def minDepth(root):
    if root is None:
        return 0
    return helper(root)

def helper(root):
    if root is None:
        return float('inf')  # 如果只有一个孩子, 返回该孩子处的高度
    if root.left is None and root.right is None:
        return 1
    l = helper(root.left)
    r = helper(root.right)
    return min(l, r) + 1
```

## Top Down

```py
class Solution:
    def minDepth(self, root):
        if root is None:
            return 0
        self.ret = float('inf')
        self.helper(root, 1)
        return self.ret

    def helper(self, root, depth):
        if root is None:
            return
        if root.left is None and root.right is None:
            self.ret = min(self.ret, depth)
        
        self.helper(root.left, depth + 1)
        self.helper(root.right, depth + 1)
        return
```
