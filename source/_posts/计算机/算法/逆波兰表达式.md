`1 2 + 3 * 4 /` 代表 `(1 + 2) * 3 / 4`.

```py
def evalRPN(tokens):
    if len(tokens) == 0:
        return 0
    stack = []
    for i in range(len(tokens)):
        if tokens[i] == '+':
            r = stack.pop()
            l = stack.pop()
            stack.append(l + r)
        elif tokens[i] == '-':
            r = stack.pop()
            l = stack.pop()
            stack.append(l - r)
        elif tokens[i] == '*':
            r = stack.pop()
            l = stack.pop()
            stack.append(l * r)
        elif tokens[i] == '/':
            r = stack.pop()
            l = stack.pop()
            stack.append(l // r)
        else:
            stack.append(int(tokens[i]))
    return stack.pop()
```
