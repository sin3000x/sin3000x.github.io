## Top Down
```py
class Solution:
    def exist(self, root, target):
        self.target = target
        self.ret = False
        self.helper(root, 0)
        return self.ret
    
    def helper(selr, root, current):
        if root is None:
            return
        if root.left is None and root.right is None:
            self.ret = self.ret or current + root.val == self.target
            return
        if self.ret == True:
            return
        self.helper(root.left, current + root.val)
        self.helper(root.right, current + root.val)
```

## Top Down + Bottom Up
```py
class Solution:
    def exist(self, root, target):
        self.target = target
        return self.helper(root, 0)

    def helper(self, root, current):
        if root is None:
            return False
        if root.left is None and root.right is None:
            return current + root.val == self.target
        
        return self.helper(root.left, current + root.val) or self.helper(root.right, current + root.val)
```
